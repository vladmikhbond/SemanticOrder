@2 Введення та виведення на консоль
@@ Console.ReadLine|Console.Write|Console.WriteLine|Convert.ToInt32|Convert.ToDouble

@3 Введення з консолі
{{
string s = Console.ReadLine();

// Перетворення на ціле число
short h = Convert.ToInt16(s);
int i = Convert.ToInt32(s);
long l = Convert.ToInt64(s);

// Перетворення на дійсне число
float f = Convert.ToSingle(s);
double d = Convert.ToDouble(s);
}}

@3 Виведення рядків
{{
int x = 10, y = 20;
Console.Write("Ширина = "+ x +" Висота = "+ y);
}}

@3 Виведення з форматуванням.
{{
Console.Write("Ширина = {0} Висота = {1}", x, y);

Console.Write($"Ширина = {x} Висота = {y}");
}}

@2 Форматування чисел
@@ -

@3 Шаблон для форматування чисел може включати додаткову інформацію. Взагалі, його cтруктура така:
{{
{ index [,alignment] [:formatString] }
}}
{index} – номер значення у списку значень;
{alignment} – ціле зі знаком – ширина поля та вирівнювання (якщо ширина додатна – вирівнювання вправо, якщо від'ємна – вліво);
{:formatString} – закодований опис перетворення.

@3 Наприклад,
{{
{0,-10:F2}
}} означає: вирівнювання вліво, ширина не менше 10 позицій, формат з фіксованою точкою, 2 знаки після коми.
